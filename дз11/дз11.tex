	1. Найти $y'(0), y > -5$\\
	
	$x ^ 2 + y ^ 2 - 6x + 5y - 14 = 0$ 
	
	$2x + 2yy' - 6 + 5y' = 0$
	
	$y'(2y + 5) = -2x + 6$
	
	$y' = \dfrac{-2x + 6}{2y + 5}$ 
	
	Найдем корни изначального уравнения при $x = 0$
	
	$y ^ 2 + 5y - 14 = 0$
	
	$(y + 7)(y - 2) = 0$
	
	$y = 2$ 
	
	Подставим $y = 2, x = 0$ в уравнение производной
	
	$
	y'(0) = \dfrac{6}{9} = \dfrac{2}{3}
	$
	
	\answer{$\dfrac{2}{3}$}  \sspace
	
	2. Найти производную
	
	$y'(x) = \dfrac{y'(t)}{x'(t)} = \dfrac{(t^3)'}{(-e^{-t})'} = \dfrac{3t^2}{-e^{-t}} = 3t^2e^t$ \sspace
	
	\answer{$y'(x) = 3t^2e^t $}
	\sspace
	
	3. Найти в точке $y_0 = 1$ производную функцию, обратную к $y(x) = e^x + x$
	
	$y(x) = e^x + x$
	
	$x = e^{y} + y$
	
	$1 - e^{y}y' -y' = 0$
	
	$y'(-e^{y} - 1) = -1$
	
	$y'(x) = \dfrac{1}{e^{y} + 1}$
	
	$y'(1) = \dfrac{1}{e + 1}$
	
	$\answer{\dfrac{1}{e + 1}$
